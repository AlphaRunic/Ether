Fundamentals

Ether is a high-level dynamically-typed interpreter. It has C-style syntax.
Some basic fundamental concepts in Ether include:

print "Hello, world!";

let foo = "bar";
for (let i = 0; i < 10; i++)
  foo = foo + " bar";

let count = 0;
while (true) {
  count = count + 1;
  if (count == 10)
    break;
}

if (count != 10)
  raise "Something went very wrong.";
else
  print "Everything is normal.";

method sayHello(name) {
  print "Hello, " + name ".";
}

sayHello("world");
sayHello("Jane");
